если делаешь какое то событие через ад листенер и нужно всплытие
обязательно посмотри всплывает ли это событие или нет 
к примеру у меня был инпут если бы я поставил событие focus 
то ни чего не сработало потому что оно не всплывает 
тут нужно было использовать focusin оно уже всплывает 

вот тот пример 

let form = document.querySelector('.form');
let btn = document.querySelector('.btn');
let inputs = document.querySelectorAll('.check');

btn.addEventListener('click', function (e) {

   for (let i = 0; i < inputs.length; i++) {
      if (inputs[i].value === '') {
         inputs[i].classList.add('err');
         e.preventDefault();
      } else {
         inputs[i].classList.remove('err');
         return false;
      }

      form.addEventListener('focusin', changeColor);  если бы тут был focus ни чего бы не работало, оно не всплывает

      function changeColor(e) {
         if (e.target.classList.contains('check')) {
            e.target.classList.remove('err');
         }
      }
      
   }

});
